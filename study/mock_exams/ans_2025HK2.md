## Câu 1 (5 điểm) — Lớp CMyArray

### Đề bài (main cho sẵn)
```cpp
int main() {
    CMyArray arr;
    arr.Add(5); arr.Add(4); arr.Add(9); arr.Add(6);
    int minIndex = arr.Min();        // trả về CHỈ SỐ phần tử nhỏ nhất
    if (minIndex != -1) {
        cout << arr[minIndex] << endl;  // in GIÁ TRỊ tại chỉ số đó
    } else {
        cout << "Mang rong." << endl;
    }
    return 0;
}
// Output: 4
```

### Phân tích yêu cầu từ main
| Lệnh trong main | Suy ra class cần |
|----------------|-----------------|
| `CMyArray arr;` | Default constructor |
| `arr.Add(5);` | `void Add(int value)` |
| `arr.Min()` → trả về int | `int Min() const` — trả về **chỉ số** (index), -1 nếu rỗng |
| `arr[minIndex]` | `int operator[](int i) const` |
| (Big Three bắt buộc vì có dynamic memory) | copy ctor + op= + dtor |

### Đáp án hoàn chỉnh

```cpp
#include <iostream>
using namespace std;

class CMyArray {
private:
    int* _data;       // mảng động
    int  _size;       // số phần tử hiện tại
    int  _capacity;   // dung lượng đã cấp phát

public:
    // ── Constructor ────────────────────────────────────────────────
    CMyArray() : _data(nullptr), _size(0), _capacity(0) {}

    // ── Destructor (Big Three #1) ───────────────────────────────────
    ~CMyArray() {
        delete[] _data;
        _data = nullptr;
    }

    // ── Copy Constructor (Big Three #2) — deep copy ────────────────
    CMyArray(const CMyArray& other) {
        _size     = other._size;
        _capacity = other._capacity;
        _data     = new int[_capacity];
        for (int i = 0; i < _size; i++)
            _data[i] = other._data[i];
    }

    // ── Assignment Operator (Big Three #3) — deep copy ────────────
    CMyArray& operator=(const CMyArray& other) {
        if (this == &other) return *this;   // kiểm tra tự gán
        delete[] _data;                      // giải phóng bộ nhớ cũ
        _size     = other._size;
        _capacity = other._capacity;
        _data     = new int[_capacity];
        for (int i = 0; i < _size; i++)
            _data[i] = other._data[i];
        return *this;
    }

    // ── Add: thêm phần tử vào cuối ────────────────────────────────
    void Add(int value) {
        if (_size == _capacity) {           // hết chỗ → mở rộng gấp đôi
            int newCap = (_capacity == 0) ? 4 : _capacity * 2;
            int* newData = new int[newCap];
            for (int i = 0; i < _size; i++)
                newData[i] = _data[i];
            delete[] _data;
            _data     = newData;
            _capacity = newCap;
        }
        _data[_size++] = value;
    }

    // ── Min: trả về CHỈ SỐ phần tử nhỏ nhất, -1 nếu rỗng ─────────
    int Min() const {
        if (_size == 0) return -1;
        int minIdx = 0;
        for (int i = 1; i < _size; i++)
            if (_data[i] < _data[minIdx])
                minIdx = i;
        return minIdx;
    }

    // ── operator[] — đọc phần tử theo chỉ số ──────────────────────
    int operator[](int i) const { return _data[i]; }
    int& operator[](int i)      { return _data[i]; }

    // ── Tiện ích ───────────────────────────────────────────────────
    int Size() const { return _size; }
};

int main() {
    CMyArray arr;
    arr.Add(5); arr.Add(4); arr.Add(9); arr.Add(6);
    int minIndex = arr.Min();
    if (minIndex != -1) {
        cout << arr[minIndex] << endl;  // → 4
    } else {
        cout << "Mang rong." << endl;
    }
    return 0;
}
```

### Giải thích điểm quan trọng

**Tại sao `Min()` trả về index, không phải value?**
Vì `arr[minIndex]` dùng index để lấy value. Nếu `Min()` trả về value thì `arr[value]` sẽ sai.

**Tại sao cần Big Three?**
Class có `delete[] _data` trong destructor → Rule of Three → phải thêm copy ctor và op=.
Nếu thiếu: `CMyArray b = arr; b[0] = 99;` sẽ làm arr[0] cũng thành 99 (shallow copy → 2 pointer cùng trỏ 1 vùng nhớ).

**Tại sao op= cần kiểm tra `if (this == &other)`?**
`arr = arr;` — nếu không kiểm tra → `delete[] _data` xóa data → copy từ data đã xóa → crash.

## Câu 2 (5 điểm) — Hệ thống thiết bị hiển thị

### Yêu cầu
- Base class: thông tin chung (tên, mã, trạng thái)
- 3 lớp con: Monitor, Projector, SmartTV — mỗi cái có thuộc tính riêng
- Phương thức: `turnOn()`, `turnOff()`, `displayContent()`, `getDeviceType()`, `displayInfo()`
- **Bắt buộc đa hình**

### Đáp án hoàn chỉnh

```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

// ══════════════════════════════════════════════════════════════════
// BASE CLASS — abstract (có pure virtual method)
// ══════════════════════════════════════════════════════════════════
class DisplayDevice {
protected:
    string _deviceName;
    string _deviceId;
    bool   _isOn;

public:
    DisplayDevice(string name, string id)
        : _deviceName(name), _deviceId(id), _isOn(false) {}

    virtual ~DisplayDevice() {}   // ← BẮT BUỘC khi dùng base pointer

    void turnOn()  { _isOn = true;  cout << _deviceName << " da bat.\n"; }
    void turnOff() { _isOn = false; cout << _deviceName << " da tat.\n"; }

    // pure virtual → bắt buộc override ở lớp con
    virtual void displayContent(const string& content) = 0;
    virtual string getDeviceType() const = 0;
    virtual void displayInfo() const = 0;
};

// ══════════════════════════════════════════════════════════════════
// LỚP CON 1 — Monitor
// ══════════════════════════════════════════════════════════════════
class Monitor : public DisplayDevice {
private:
    string _resolution;    // "1920x1080"
    int    _refreshRateHz;

public:
    Monitor(string name, string id, string resolution, int refreshRate)
        : DisplayDevice(name, id),
          _resolution(resolution), _refreshRateHz(refreshRate) {}

    void displayContent(const string& content) override {
        cout << "Man hinh '" << _deviceName
             << "' dang hien thi: '" << content << "'\n";
    }

    string getDeviceType() const override { return "Monitor"; }

    void displayInfo() const override {
        cout << "=== " << getDeviceType() << " ===\n"
             << "  Ten: " << _deviceName << "\n"
             << "  Ma:  " << _deviceId   << "\n"
             << "  Trang thai: " << (_isOn ? "Bat" : "Tat") << "\n"
             << "  Do phan giai: " << _resolution << "\n"
             << "  Tan so quet: " << _refreshRateHz << " Hz\n";
    }
};

// ══════════════════════════════════════════════════════════════════
// LỚP CON 2 — Projector
// ══════════════════════════════════════════════════════════════════
class Projector : public DisplayDevice {
private:
    int    _lumens;
    double _throwRatio;

public:
    Projector(string name, string id, int lumens, double throwRatio)
        : DisplayDevice(name, id),
          _lumens(lumens), _throwRatio(throwRatio) {}

    void displayContent(const string& content) override {
        cout << "May chieu '" << _deviceName
             << "' dang trinh chieu: '" << content
             << "' (do sang " << _lumens << " lumens).\n";
    }

    string getDeviceType() const override { return "Projector"; }

    void displayInfo() const override {
        cout << "=== " << getDeviceType() << " ===\n"
             << "  Ten: " << _deviceName << "\n"
             << "  Ma:  " << _deviceId   << "\n"
             << "  Trang thai: " << (_isOn ? "Bat" : "Tat") << "\n"
             << "  Do sang: " << _lumens << " lumens\n"
             << "  Ti le chieu: " << _throwRatio << "\n";
    }
};

// ══════════════════════════════════════════════════════════════════
// LỚP CON 3 — SmartTV
// ══════════════════════════════════════════════════════════════════
class SmartTV : public DisplayDevice {
private:
    double _screenSizeInches;
    string _operatingSystem;
    int    _currentChannel;

public:
    SmartTV(string name, string id, double screenSize, string os)
        : DisplayDevice(name, id),
          _screenSizeInches(screenSize), _operatingSystem(os),
          _currentChannel(1) {}

    void displayContent(const string& content) override {
        cout << "Tivi thong minh '" << _deviceName
             << "' dang xem: '" << content
             << "' (Kenh: " << _currentChannel << ").\n";
    }

    string getDeviceType() const override { return "SmartTV"; }

    void setChannel(int ch) { _currentChannel = ch; }

    void displayInfo() const override {
        cout << "=== " << getDeviceType() << " ===\n"
             << "  Ten: " << _deviceName << "\n"
             << "  Ma:  " << _deviceId   << "\n"
             << "  Trang thai: " << (_isOn ? "Bat" : "Tat") << "\n"
             << "  Kich thuoc: " << _screenSizeInches << " inch\n"
             << "  He dieu hanh: " << _operatingSystem << "\n"
             << "  Kenh hien tai: " << _currentChannel << "\n";
    }
};

// ══════════════════════════════════════════════════════════════════
// CONTAINER — quản lý danh sách thiết bị
// ══════════════════════════════════════════════════════════════════
class DeviceManager {
private:
    vector<DisplayDevice*> _devices;

public:
    ~DeviceManager() {
        for (auto d : _devices) delete d;   // giải phóng bộ nhớ
    }

    void addDevice(DisplayDevice* d) { _devices.push_back(d); }

    void showAllInfo() const {
        for (auto d : _devices) d->displayInfo();   // virtual dispatch
    }

    void showAllContent(const string& content) const {
        for (auto d : _devices) d->displayContent(content);
    }
};

// ══════════════════════════════════════════════════════════════════
// MAIN
// ══════════════════════════════════════════════════════════════════
int main() {
    DeviceManager mgr;
    mgr.addDevice(new Monitor("Dell U2722D", "M001", "2560x1440", 60));
    mgr.addDevice(new Projector("Epson EB-X", "P001", 3600, 1.2));
    mgr.addDevice(new SmartTV("Samsung 55\"", "T001", 55.0, "Android TV"));

    mgr.showAllInfo();
    cout << "\n--- Phat noi dung ---\n";
    mgr.showAllContent("Bai giang OOP");
    return 0;
}
```

### 3 điều không được quên trong câu 2

```
1. virtual ~DisplayDevice() {}          ← KHÔNG CÓ → memory leak khi delete base pointer
2. virtual void displayContent() = 0   ← KHÔNG CÓ → không bắt buộc lớp con implement
3. for (auto d : _devices) delete d    ← KHÔNG CÓ trong ~DeviceManager → memory leak
```

## Tổng kết điểm mạnh của đề này

| Kỹ năng kiểm tra | Câu |
|-----------------|-----|
| Big Three (Rule of Three) | 1 |
| Dynamic memory resize | 1 |
| operator[] const + non-const | 1 |
| Pure virtual method | 2 |
| Virtual destructor | 2 |
| vector<Base*> + delete | 2 |
| Override đúng chữ ký | 2 |
