## Câu 1 (5 điểm) — Lớp CMyString

### Đề bài (main cho sẵn)
```cpp
int main() {
    CMyString s1("ABC.");
    s1.Append("DEE.");
    // s1 = "ABC.DEE."
    int foundIndex = s1.Find("DE");   // → tìm "DE" trong "ABC.DEE." → vị trí 4
    if (foundIndex != -1) {           // NOTE: đề gốc có lỗi điều kiện, logic đúng là != -1
        cout << "Tim thay chuoi con tai vi tri: " << foundIndex;
        cout << "Ky tu tai vi tri do la: " << s1[foundIndex];
    } else {
        cout << "Khong tim thay chuoi con.";
    }
    CMyString s2("C++");
    CMyString s3 = s1 + s2;   // s3 = "ABC.DEE.C++"
    return 0;
}
// Output:
// Tim thay chuoi con tai vi tri: 4
// Ky tu tai vi tri do la: D
```

### Phân tích yêu cầu từ main

| Lệnh | Suy ra |
|------|--------|
| `CMyString s1("ABC.")` | Constructor nhận `const char*` |
| `s1.Append("DEE.")` | `void Append(const char*)` — nối thêm chuỗi |
| `s1.Find("DE")` | `int Find(const char*)` — trả về index, -1 nếu không tìm thấy |
| `s1[foundIndex]` | `char operator[](int i) const` |
| `CMyString s3 = s1 + s2` | `CMyString operator+(const CMyString&) const` |
| (có dynamic `char*` bên trong) | **Big Three** bắt buộc |

### Đáp án hoàn chỉnh

```cpp
#include <iostream>
#include <cstring>   // strlen, strcpy, strcat, strstr
using namespace std;

class CMyString {
private:
    char* _data;   // chuỗi ký tự động (null-terminated)
    int   _len;    // độ dài (không tính '\0')

    // Helper: cấp phát và copy
    void allocAndCopy(const char* src) {
        _len  = strlen(src);
        _data = new char[_len + 1];
        strcpy(_data, src);
    }

public:
    // ── Constructor nhận const char* ────────────────────────────
    CMyString(const char* str = "") {
        allocAndCopy(str);
    }

    // ── Destructor (Big Three #1) ────────────────────────────────
    ~CMyString() {
        delete[] _data;
        _data = nullptr;
    }

    // ── Copy Constructor (Big Three #2) — deep copy ──────────────
    CMyString(const CMyString& other) {
        allocAndCopy(other._data);
    }

    // ── Assignment Operator (Big Three #3) ───────────────────────
    CMyString& operator=(const CMyString& other) {
        if (this == &other) return *this;
        delete[] _data;
        allocAndCopy(other._data);
        return *this;
    }

    // ── Append: nối chuỗi vào cuối ───────────────────────────────
    void Append(const char* suffix) {
        int newLen   = _len + strlen(suffix);
        char* newData = new char[newLen + 1];
        strcpy(newData, _data);
        strcat(newData, suffix);
        delete[] _data;
        _data = newData;
        _len  = newLen;
    }

    // ── Find: tìm chuỗi con, trả về index hoặc -1 ────────────────
    int Find(const char* sub) const {
        char* pos = strstr(_data, sub);   // strstr trả về pointer đến vị trí tìm thấy
        if (pos == nullptr) return -1;
        return (int)(pos - _data);        // chênh lệch pointer = index
    }

    // ── operator[]: đọc ký tự theo chỉ số ────────────────────────
    char operator[](int i) const { return _data[i]; }
    char& operator[](int i)      { return _data[i]; }

    // ── operator+: ghép 2 chuỗi → chuỗi mới ─────────────────────
    CMyString operator+(const CMyString& other) const {
        int newLen    = _len + other._len;
        char* newData = new char[newLen + 1];
        strcpy(newData, _data);
        strcat(newData, other._data);
        CMyString result(newData);
        delete[] newData;
        return result;
    }

    // ── Tiện ích ─────────────────────────────────────────────────
    int Length() const { return _len; }

    void Print() const { cout << _data; }
};

int main() {
    CMyString s1("ABC.");
    s1.Append("DEE.");
    // s1 = "ABC.DEE."

    int foundIndex = s1.Find("DE");  // → 4
    if (foundIndex != -1) {
        cout << "Tim thay chuoi con tai vi tri: " << foundIndex << "\n";
        cout << "Ky tu tai vi tri do la: " << s1[foundIndex] << "\n";
    } else {
        cout << "Khong tim thay chuoi con.\n";
    }

    CMyString s2("C++");
    CMyString s3 = s1 + s2;  // s3 = "ABC.DEE.C++"
    s3.Print();
    return 0;
}
```

### Giải thích các điểm kỹ thuật

**`strstr` dùng như thế nào?**
```cpp
char* pos = strstr("ABC.DEE.", "DE");
// pos trỏ đến ký tự 'D' trong "ABC.DEE."
// pos - _data = địa chỉ của 'D' - địa chỉ đầu chuỗi = index = 4
```

**Tại sao `operator+` cần cấp phát `newData` rồi `delete`?**
```cpp
CMyString operator+(const CMyString& other) const {
    char* newData = new char[newLen + 1];  // tạm
    // ...
    CMyString result(newData);  // constructor copy newData vào result._data
    delete[] newData;           // xóa tạm
    return result;              // trả về (copy constructor được gọi)
}
```
Nếu không xóa `newData` → memory leak.

**Big Three trong CMyString — vì sao cần?**
```
Có delete[] _data trong dtor  →  phải có copy ctor (deep copy char array)
                               →  phải có op= (deep copy + self-check)
```

## Câu 2 (5 điểm) — Quản lý phương tiện giao thông

### Yêu cầu

| Loại xe | Phí cầu đường/năm |
|---------|------------------|
| Ô tô | 1.5 triệu + (nếu ≥ 7 chỗ: thêm 0.5 triệu) |
| Xe máy | Cố định 0.5 triệu |
| Xe tải | 5 triệu + 2 triệu × tải trọng tối đa (tấn) |

### Đáp án hoàn chỉnh

```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

// ══════════════════════════════════════════════════════════════════
// BASE CLASS
// ══════════════════════════════════════════════════════════════════
class PhuongTien {
protected:
    string _tenXe;
    string _bienSo;

public:
    PhuongTien(string tenXe, string bienSo)
        : _tenXe(tenXe), _bienSo(bienSo) {}

    virtual ~PhuongTien() {}

    // pure virtual → bắt buộc lớp con tính phí theo quy tắc riêng
    virtual double tinhPhiCauDuong() const = 0;

    virtual void xuatThongTin() const {
        cout << "  " << _tenXe << " [" << _bienSo << "]"
             << " - Phi: " << tinhPhiCauDuong() << " trieu/nam\n";
    }
};

// ══════════════════════════════════════════════════════════════════
// LỚP CON 1 — Ô tô
// ══════════════════════════════════════════════════════════════════
class OTo : public PhuongTien {
private:
    int _soChoNgoi;

public:
    OTo(string tenXe, string bienSo, int soChoNgoi)
        : PhuongTien(tenXe, bienSo), _soChoNgoi(soChoNgoi) {}

    double tinhPhiCauDuong() const override {
        double phi = 1.5;
        if (_soChoNgoi >= 7) phi += 0.5;
        return phi;
    }
};

// ══════════════════════════════════════════════════════════════════
// LỚP CON 2 — Xe máy
// ══════════════════════════════════════════════════════════════════
class XeMay : public PhuongTien {
public:
    XeMay(string tenXe, string bienSo)
        : PhuongTien(tenXe, bienSo) {}

    double tinhPhiCauDuong() const override {
        return 0.5;   // cố định
    }
};

// ══════════════════════════════════════════════════════════════════
// LỚP CON 3 — Xe tải
// ══════════════════════════════════════════════════════════════════
class XeTai : public PhuongTien {
private:
    double _taiTrongToiDa;   // đơn vị: tấn

public:
    XeTai(string tenXe, string bienSo, double taiTrong)
        : PhuongTien(tenXe, bienSo), _taiTrongToiDa(taiTrong) {}

    double tinhPhiCauDuong() const override {
        return 5.0 + 2.0 * _taiTrongToiDa;
    }
};

// ══════════════════════════════════════════════════════════════════
// CONTAINER — quản lý đội xe
// ══════════════════════════════════════════════════════════════════
class CongTyVanTai {
private:
    vector<PhuongTien*> _dsPhuongTien;

public:
    ~CongTyVanTai() {
        for (auto p : _dsPhuongTien) delete p;
    }

    void them(PhuongTien* pt) { _dsPhuongTien.push_back(pt); }

    double tinhTongPhi() const {
        double tong = 0;
        for (auto p : _dsPhuongTien)
            tong += p->tinhPhiCauDuong();   // virtual dispatch
        return tong;
    }

    void xuatDanhSach() const {
        cout << "=== Danh sach phuong tien ===\n";
        for (auto p : _dsPhuongTien)
            p->xuatThongTin();
        cout << "Tong phi cau duong: " << tinhTongPhi() << " trieu VND/nam\n";
    }
};

// ══════════════════════════════════════════════════════════════════
// MAIN
// ══════════════════════════════════════════════════════════════════
int main() {
    CongTyVanTai cty;
    cty.them(new OTo("Toyota Fortuner", "51A-123", 7));   // phi = 2.0
    cty.them(new OTo("Honda City", "51B-456", 5));         // phi = 1.5
    cty.them(new XeMay("Honda Wave", "59X1-789"));         // phi = 0.5
    cty.them(new XeTai("Isuzu 5T", "51C-321", 5.0));       // phi = 15.0
    cty.them(new XeTai("Hino 2T", "51D-654", 2.0));        // phi = 9.0

    cty.xuatDanhSach();
    // Tong: 2.0 + 1.5 + 0.5 + 15.0 + 9.0 = 28.0 trieu/nam
    return 0;
}
```

### Kiểm tra lại phép tính
| Xe | Tính | Kết quả |
|----|------|---------|
| Fortuner 7 chỗ | 1.5 + 0.5 | 2.0 triệu |
| Honda City 5 chỗ | 1.5 | 1.5 triệu |
| Honda Wave | cố định | 0.5 triệu |
| Isuzu 5 tấn | 5 + 2×5 | 15.0 triệu |
| Hino 2 tấn | 5 + 2×2 | 9.0 triệu |
| **Tổng** | | **28.0 triệu** |

## Tổng kết — Điểm phân biệt CMyString với CMyArray

| | CMyArray | CMyString |
|---|---|---|
| Kiểu dữ liệu | `int*` | `char*` |
| Khởi tạo | `_data = new int[cap]` | `_data = new char[len+1]` |
| Copy | vòng for | `strcpy` |
| Nối | resize mảng | `strcat` / `Append` |
| Tìm | tự viết vòng for | `strstr` |
| Kết thúc | không cần `'\0'` | **bắt buộc** null-terminate |
